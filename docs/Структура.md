```text
. 
├── reshala.sh              # Главный скрипт. Точка входа, отвечает за запуск меню и роутинг к модулям.
|
├── config/
│   └── reshala.conf        # Конфигурация. Все постоянные настройки и переменные окружения.
|
├── install.sh              # Скрипт для первичной установки Решалы на сервер.
|
├── modules/                # Основные модули системы. Здесь живут крупные блоки функционала.
│   ├── core/               # Модули ядра: общие хелперы, генератор меню, обновление, сканер состояния.
│   │   ├── common.sh       # Общие функции: цвета, логирование, UI-хелперы, run_cmd, ввод/вывод.
│   │   ├── dependencies.sh # Хелпер для установки системных зависимомостей (curl, jq, etc.).
│   │   ├── menu_generator.sh # Сканирует манифесты @menu.manifest и строит все меню в системе.
│   │   ├── self_update.sh  # Логика самообновления и деинсталляции скрипта.
│   │   └── state_scanner.sh # Определяет наличие Remnawave-окружения и другие системные статусы.
│   │
│   ├── security/           # Модули для усиления безопасности сервера.
│   │   ├── backup.sh       # Бэкап/восстановление конфигураций безопасности.
│   │   ├── fail2ban.sh     # Управление Fail2Ban.
│   │   ├── firewall.sh     # Управление файрволом (UFW).
│   │   ├── geoblock.sh     # Блокировка доступа по странам (IPset).
│   │   ├── kernel.sh       # Применение безопасных настроек ядра.
│   │   ├── ssh_manager.sh  # Управление портами и доступом по SSH.
│   │   ├── whitelist_manager.sh # Глобальный Белый Список (Master Whitelist).
│   │   └── status.sh       # Отображение сводного статуса безопасности.
│   │
│   ├── utils/              # Системные утилиты и механизмы обслуживания.
│   │   ├── migrations.sh   # Движок критических обновлений (Migration Engine).
│   │   └── migrations/     # Модульные патчи (0001_..., 0002_... и т.д.).
│   │
│   ├── local/              # Модули для управления локальным сервером.
│   │   ├── diagnostics.sh  # Диагностика: просмотр логов, управление Docker.
│   │   ├── local_care.sh   # Обслуживание: обновление системы, настройка сети, speedtest.
│   │   └── traffic_limiter.sh # Шейпер трафика для ограничения скорости по IP.
│   │
│   ├── skynet/             # Модули для управления флотом серверов через Skynet.
│   │   ├── db.sh           # Работа с базой данных флота.
│   │   ├── executor.sh     # Выполнение команд/плагинов на удаленных серверах.
│   │   ├── keys.sh         # Управление SSH-ключами для Skynet.
│   │   └── menu.sh         # Меню управления флотом Skynet.
│   │
│   ├── remnawave/          # Модули для установки и управления Remnawave.
│   │   └── menu.sh         # Меню Remnawave.
│   │
│   ├── telegram/           # Модули для интеграции с Telegram.
│   │   └── menu.sh         # Меню Telegram-уведомлений.
│   │
│   └── ui/                 # Модули, связанные с отрисовкой пользовательского интерфейса.
│       ├── dashboard.sh    # Отрисовка главного дашборда.
│       └── widget_manager.sh # Управление виджетами дашборда.
|
└── plugins/                # Плагины для расширения функционала.
    ├── dashboard_widgets/  # Виджеты для главного дашборда.
    └── skynet_commands/    # Команды, выполняемые на удалённых серверах через Skynet.
```
